#include <iostream>
using namespace std;
//Im creating a function that inputs grades
void inputGrades(double& g1, double& g2, double& g3){

    cin >> g1;
    cin >> g2;
    cin >> g3;

}

// creating a function to calculate the average

double calculateAverage(const double g1, const double g2, const double g3){
    return (g1 + g2 + g3) /3.0;

}

//  takes the average grade and return letter
char getLetterGrade(const double average){
    if (average >= 90 ){
        return 'A';
    }else if (average >= 80){
        return 'B';
    }else if (average >= 70){
        return 'C';

    }else if (average >= 60){
        return 'D';

    } else {
        return 'F';
    }
}
int main() {
    string pass;
    do{
        double grade1;
        double grade2;
        double grade3;
        double tot;
        char ot;

        inputGrades(grade1, grade2, grade3);
        tot = calculateAverage(grade1, grade2, grade3);
        ot = getLetterGrade(tot);
        cout << tot <<  "-" << ot << endl;

        switch (ot) {
            case 'A': cout << "That's Great!" << endl; break;
                case 'B': cout << "That's good" << endl; break;
                case 'C': cout << "I mean you did pass" << endl; break;
                case 'D': cout << "Come on, do better" << endl; break;
            case 'F':
                cout << "Aw man!" << endl;
                break;

        }
        cout << "Did you Pass?" << endl;
        cin >> pass;
    }while (pass == "yes");
    cout << "I know you got it next time!" << endl;
    return 0;


}
